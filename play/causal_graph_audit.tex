    {shape = rectangle,
    rounded corners,

    \node [right = 2 of B] (P_G) {External mechanism ($p_G$)};
    
    % Equilibrium
    \node [right = 1 of B, below = 1 of B] (alpha) {Audit effort};
    \node [right = 1 of alpha] (beta) {Attempted misstatement};

    \node [below = 1 of beta] (M) {Misstatement};

    \node [right = 1 of M] (R) {Restatement};

    \draw (P_G) edge[->] (alpha);
    
    \draw (alpha) edge[->] (M);
    \draw (beta) edge[->] (M);
